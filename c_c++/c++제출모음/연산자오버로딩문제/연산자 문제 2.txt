#include <iostream>
#include <Windows.h>
using namespace std;

/*
Time Class를 만들어 객체 + 객체 가 가능한 연산자 오버로딩을 작성 후
프로그램을 완성하시오.
*/

class Time
{
private:
	int m_iHour;
	int m_iMin;
	int m_iDay;
public:
	Time() { m_iHour = 0, m_iMin = 0, m_iDay = 0; };
	void TimeSet(int Hour, int Min) { m_iHour = Hour; m_iMin = Min; }
	friend Time operator + (Time main, Time tmp);
	void ShowTime()
	{
		cout << "총 공부 시간 : " << m_iHour << " : " << m_iMin << endl;
		cout << "===공부 시간 관리 프로그램 < " << m_iDay << "Day>===" << endl;
		cout << "\t1.시간 등록" << endl;
		cout << "\t2.종료" << endl;
		cout << "\t입력";
	}


};

Time operator + (Time main, Time tmp)
{
	main.m_iHour += tmp.m_iHour;
	main.m_iMin += tmp.m_iMin;
	while (main.m_iMin >= 60)
	{
		main.m_iMin -= 60;
		main.m_iHour++;
	}
	++main.m_iDay;
	return main;
}

void main()
{
	Time T, T2;
	int Select;
	int tmp1, tmp2;
	while (1)
	{
		system("cls");
		T.ShowTime();
		cin >> Select;
		switch (Select)
		{
		case 1:
			cout << "시간 : ";
			cin >> tmp1;
			cout << "\t분 : ";
			cin >> tmp2;
			T2.TimeSet(tmp1, tmp2);
			T = T + T2;
			break;
		case 2:
			return;
		}
	}
}
