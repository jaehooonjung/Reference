#include<stdio.h>

int main(void) {
	int n1, n2, k, m, e;
	printf("1.정수를 입력하시오 : ");
	scanf("%d", &n1);
	(n1 % 5 == 0) ? printf("%d는(은) 5의배수 입니다.\n", n1) : printf("%d는(은) 5의 배수가 아닙니다.\n", n1);

	printf("2. 세 과목의 점수를 입력하시오\n");
	printf("국어 점수 : ");
	scanf("%d", &k);
	printf("수학 점수 : ");
	scanf("%d", &m);
	printf("영어 점수 : ");
	scanf("%d", &e);
	printf("평균 %.2f", (float)(k + m + e) / 3); //float형 계산에 3을 3.0으로 써서 실수형 맞춰주기
	((float)(k + m + e) / 3 < 60) ? printf("불합격\n") : printf("합격\n"); //평균값 비교하는 부등호에 <=으로 바꿔주기 평균이 60이 나올면 합격이라 뜨지만 오류때문에

	printf("3. 두 정수를 입력하시오 : ");
	scanf("%d%d", &n1, &n2);
	(n1>n2) ? printf("큰 수 : %d\n", n1) : printf("큰 수 : %d\n", n2);

	printf("4.정수를 입력하시오 : ");
	scanf("%d", &n1);
	(n1 % 3 == 0 && n1 % 2 == 0) ? printf("%d는 짝수이며 3의배수 입니다..\n", n1) : printf("조건실패\n");

	printf("5.정수를 입력하시오 : ");
	scanf("%d", &n1);
	(n1 % 5 == 0 || n1 % 7 == 0) ? printf("%d는 5의 배수이거나 7의배수 입니다.\n", n1) : printf("조건실패\n");
	return 0;
}
