#include<stdio.h>

void main() {
	int n1, n2, n3, m, n;
	printf("정수 세개를 입력하시오 : ");
	scanf("%d%d%d", &n1, &n2, &n3);

	if (n1>=n2&&n1>=n3)
	{
		m = n1;
		if (n2<=n3)
		{
			n = n2;
		}
		else
		{
			n = n3;
		}
	}

	if (n2>=n1&&n2>=n3)
	{
		m = n2;
		if (n1<=n3)
		{
			n = n1;
		}
		else
		{
			n = n3;
		}
	}

	if (n3>=n1&&n3>=n2)
	{
		m = n3;
		if (n1<=n2)
		{
			n = n1;
		}
		else
		{
			n = n2;
		}
	}


	printf("\n max = %d min=%d ", m, n);
}
