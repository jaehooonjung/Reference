#include<stdio.h>
#define SIZE 10

void main()
{
	int arr[SIZE] = { 10,17,3,9,37,10,8,9,13,21 };
	printf("1번 문제\n");
	for (int i = 0; i < SIZE; i++)
		printf("%d,", arr[i]);

	printf("\b \n2번 문제\n");
	int invert_arr[SIZE];
	for (int i = 0, tmp = 9; i < SIZE; i++, tmp--)
		invert_arr[i] = arr[tmp];
	printf("invert_arr : ");
	for (int i = 0; i < SIZE; i++)
		printf("%d,", invert_arr[i]);
	printf("\b \narr : ");
	for (int i = 0; i < SIZE; i++)
		printf("%d,", arr[i]);
	printf("\b \n");

	printf("\b 3번 문제\n");
	int odd_sum = 0, x2_sum = 0, count1 = 0, count2 = 0, odd_arr[SIZE], x2_arr[SIZE];
	for (int i = 0; i < SIZE; i++)
	{
		if (i % 2 == 1)
		{
			odd_sum += arr[i];
			odd_arr[count1] = arr[i];
			count1++;
		}
		else
			x2_sum += arr[i];
		x2_arr[count2] = arr[i];
		count2++;
	}
	printf("홀수 번째 원소 : ");
	for (int i = 0; i < 5; i++)
		printf("%d,", odd_arr[i]);
	printf("\b \n홀수 번째 총합 : %d\n", odd_sum);

	printf("짝수 번째 원소 : ");
	for (int i = 0; i < 5; i++)
		printf("%d,", x2_arr[i]);
	printf("\b \n짝수 번째 총합 : %d\n", x2_sum);

	printf("4번 문제\narr : ");
	for (int i = 0; i < SIZE; i++)
		printf("%d,", arr[i]);

	int sort_arr[SIZE];
	int test,tmp1,tmp2;
	for (int i = 0; i < SIZE; i++)
	{
		tmp1 = arr[i];
		for (int j = 0; j < SIZE; j++)
		{
			test = 1;
			tmp2 = 0;
			if (tmp1 <= arr[j])
			{
				while (i > tmp2)
				{
					if (sort_arr[tmp2] <= arr[j])
						test = 0;

					tmp2++;
				}
				if (test == 0)
					continue;
				else
				tmp1 = arr[j];
			}
		}
		sort_arr[i] = tmp1;
	}


	printf("\b \nsort_arr : ");
	for (int i = 0; i < SIZE; i++)
		printf("%d,", sort_arr[i]);
}
