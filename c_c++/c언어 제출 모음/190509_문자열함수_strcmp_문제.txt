#include<stdio.h>
#include<string.h>

int S_Compare(char str1[], char str2[])
{
	for (int i = 0; (str1[i] != NULL) || (str2[i] != NULL); i++)
	{
		if (str1[i] != str2[i])
		{
			if (str1[i] < str2[i])
				return -1;
			else if (str1[i] > str2[i])
				return 1;
		}
	}	
	return 0;

}

void main()
{
	char str1[10] = "string!!";
	char str2[10] = "string";

	printf("%s == %s : %d\n", str1, str2, S_Compare(str1, str2));
}
