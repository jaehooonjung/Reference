typedef struct node{
	char name[20];
	char add[20];
	int sum,kor,eng,mat,age;
	char abs;
	struct node * link;
}node;
node * get_node(){
	node * tmp;
	tmp=(node*)malloc(sizeof(node));
	tmp->link=0;
	return tmp;
}
void insert(node ** head){
	if(!(*head)){
		*head=get_node();
		printf("학생 이름 입력 : ");
		fflush(stdin);
		gets((*head)->name);
		printf("%s 학생 나이 입력 : ",(*head)->name);
		scanf("%d",&(*head)->age);
		fflush(stdin);
		printf("%s 학생 주소 입력 : ",(*head)->name);
		gets((*head)->add);
		printf("%s 학생 국어 입력 : ",(*head)->name);
		scanf("%d",&(*head)->kor);
		printf("%s 학생  영어 입력 : ",(*head)->name);
		scanf("%d",&(*head)->eng);
		printf("%s 학생 수학 입력 : ",(*head)->name);
		scanf("%d",&(*head)->mat);
		return ;
	}
	insert(&(*head)->link);
}
node * search(node * head,char *name){
	system("cls");
	if(!head){
		printf("찾는학생 없음 \n");
		return 0;
	}
	if(!(strcmp(head->name,name))){
		printf("정보 검색 완료\n");
		printf("========================\n");
		printf("학생 이름 : %s\n",head->name);
		printf("학생 나이 : %d 세\n",head->age);
		printf("학생 주소 : %s\n",head->add);
		printf("국어 점수 : %d\n",head->kor);
		printf("수학 점수 : %d\n",head->mat);
		head->sum=(head->eng)+(head->kor)+(head->mat);
		printf("합계 점수 : %d\n",head->sum);
		printf("평균 점수 : %.2f\n",head->sum/3.0);
		head->abs=head->sum/3;
		if(head->abs>90)
			printf("A 등급\n");
		else if(head->abs>80)
			printf("B 등급\n");
		else if(head->abs>70)
			printf("C 등급\n");
		else if(head->abs>60)
			printf("D 등급\n");
		else 
			printf("F 등급\n");
		return head;
	}
	search(head->link,name);
}
void print(node *head){
	if(!head){
		printf("\n<학생 개개인의 점수는 검색을 이용하세요 !!>\n");
		return ;
	}
	else{
		
		printf("%10s%10d%10s\n",head->name,head->age,head->add);
	}
	print(head->link);
}

void del(node ** head,char * name){
	node * tmp;
	if(!*head){
		printf("삭제할 사람 없음 \n");return ;
	}
	if(!(strcmp((*head)->name,name))){
		tmp=*head;
		*head=(*head)->link;
		free(tmp);
		printf("삭제완료\n");return ;
	}
	del(&(*head)->link,name);
}
